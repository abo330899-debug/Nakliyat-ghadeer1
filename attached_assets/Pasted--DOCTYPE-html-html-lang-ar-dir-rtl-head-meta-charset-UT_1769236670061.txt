    function parseMoney(text) {

    function formatMoney(n) {
      const sym = getCurrencySymbol();
      const s = (Number(n) || 0).toFixed(0);
      const withCommas = s.replace(/\B(?=(\d{3})+(?!\d))/g, ",");
      return withCommas + sym;
